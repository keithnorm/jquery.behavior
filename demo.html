 $.behavior('demo.colorChangeBehavior', {
  _init: function() {
    this.element //the bound element
  },

  onclick: function(e) {
    e.preventDefault();
    this.element.css({color: '#' + this._randomHex()});
  },

  //private
  _randomHex: function() {
    return Math.floor(Math.random()*16777215).toString(16);
  }
  
});      

$('a').colorChangeBehavior();
